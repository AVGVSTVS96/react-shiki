import { Card, CardGrid } from '@astrojs/starlight/components';

## Features

<CardGrid stagger>
  <Card title="Component & Hook" icon="puzzle">
    Use the `ShikiHighlighter` component for quick setup or the `useShikiHighlighter` hook for more flexibility.
  </Card>
  <Card title="Flexible Output" icon="setting">
    Choose between React elements (no `dangerouslySetInnerHTML`) or HTML strings for better performance.
  </Card>
  <Card title="Multiple Bundles" icon="seti:html">
    Full bundle (~1.2MB gz), web bundle (~695KB gz), or minimal core bundle for fine-grained control.
  </Card>
  <Card title="Custom Themes & Languages" icon="star">
    Full support for custom TextMate themes and languages.
  </Card>
  <Card title="Streaming Support" icon="rocket">
    Performant highlighting of streamed code with optional throttling.
  </Card>
  <Card title="Line Numbers" icon="list-format">
    Optional CSS-based line numbers with customizable styling.
  </Card>
</CardGrid>

## Quick Example

```tsx
import ShikiHighlighter from "react-shiki";

function CodeBlock() {
  return (
    <ShikiHighlighter language="jsx" theme="ayu-dark">
      {`const greeting = "Hello, World!";`}
    </ShikiHighlighter>
  );
}
```
